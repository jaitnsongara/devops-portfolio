Implementation Example

Comprehensive Security Scanning Script

#!/bin/bash
# Trivy security scanning automation

# Scan filesystem for vulnerabilities
trivy fs --severity HIGH,CRITICAL \
    --format json \
    --output filesystem-report.json .

# Scan Docker image
trivy image --severity HIGH,CRITICAL \
    --format json \
    --output image-report.json app:latest

# Scan Infrastructure as Code
trivy config --severity HIGH,CRITICAL \
    --format json \
    --output iac-report.json terraform/

# Generate summary report
echo "Security Scan Complete"
echo "Reports generated: filesystem, image, iac"
echo "Review reports for vulnerabilities"
